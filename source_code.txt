def main():
    x = 10
    y = 20

    if x > y:
        print("x is greater than y")
    else:
        print("x is not greater than y")

    while x > 0:
        print(x)
        x = x - 1

    for i in range(5):
        print(i)

    return

def add(a, b):
    return a + b

def multiply(a, b):
    return a * b

def factorial(n):
    if n == 0:
        return 1
    else:
        return n * factorial(n - 1)

main()
